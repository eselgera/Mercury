        Sub connection_sql()
            Try
            con = New OdbcConnection("DSN=FIPAAPP_DB;")
            If con.State = ConnectionState.Closed Then
                    con.Open()
                    'MsgBox("Conexión exitosa")
                End If
            Catch ex As Exception
                MsgBox("Error al tratar de conectar a la base de datos: " & ex.Message)
            End Try
        End Sub
